ag_imp_data_filtered <- filter(imp_data, site == "Agriculture") %>%
  mutate(`Terrain Attribute` = reorder_within(`Terrain Attribute`, `%IncMSE`, property))

# Create the plot with properly ordered x-axis in each facet
